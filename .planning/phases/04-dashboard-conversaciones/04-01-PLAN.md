### T3: conversations.service.ts — método reply
```typescript
async reply(tenantId: string, conversationId: string, body: string) {
  const conv = await this.prisma.conversation.findFirst({
    where: { id: conversationId, tenantId },
    include: { debtor: true },
  });
  if (!conv) throw new NotFoundException();
  if (conv.channel !== 'whatsapp') throw new BadRequestException('Solo WhatsApp soportado para respuesta manual');

  const phone = phonesFromDebtor(conv.debtor.phones)[0];
  if (!phone) throw new BadRequestException('Deudor sin teléfono');

  // Enviar por WhatsApp
  const result = await this.whatsapp.sendTemplate({
    to: phone,
    template_id: 'manual_reply',
    variables: { body },
    tenant_id: tenantId,
  });

  // Guardar en BD
  await this.prisma.message.create({
    data: {
      tenantId,
      conversationId,
      direction: 'out',
      channel: 'whatsapp',
      content: JSON.stringify({ text: body, human_sent: true }),
      status: result.status === 'sent' ? 'sent' : 'failed',
      sentAt: new Date(),
    },
  });

  // Si había escalación, resolverla automáticamente
  if (conv.status === 'escalated') {
    await this.prisma.conversation.update({
      where: { id: conversationId },
      data: { status: 'open' },
    });
  }

  return { sent: result.status === 'sent' };
}
```
